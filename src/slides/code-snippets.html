List<Contact> contacts = [ SELECT Id, AccountId FROM Contact ];

  Map<Id, List<Contact>> contactsByAccountIds = new Map<Id, List<Contact>>();
  for (Contact contact : contacts) {
      if (contactsByAccountIds.containsKey(contact.AccountId)) {
          List<Contact> existing = contactsByAccountIds.get(contact.AccountId);
          existing.add(contact);
          contactsByAccountIds.put(contact.AccountId, existing);
      } else {
          List<Contact> newList = new List<Contact>();
          newList.add(contact);
          contactsByAccountIds.put(contact.AccountId, newList);
      }
  }
